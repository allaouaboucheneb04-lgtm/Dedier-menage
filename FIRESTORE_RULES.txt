rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function myUser() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && myUser().data.role == "admin";
    }

    function isEmploye() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && myUser().data.role == "employe";
    }

    match /demandes_soumission/{id} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }

    match /users/{uid} {
      allow read: if isAdmin() || (signedIn() && request.auth.uid == uid);

      allow create, update, delete: if isAdmin()
        || (
          signedIn()
          && request.auth.uid == uid
          && request.resource.data.role == "employe"
          && request.resource.data.email == request.auth.token.email
          && exists(/databases/$(database)/documents/invites/$(request.resource.data.inviteCode))
          && get(/databases/$(database)/documents/invites/$(request.resource.data.inviteCode)).data.email == request.auth.token.email
          && get(/databases/$(database)/documents/invites/$(request.resource.data.inviteCode)).data.status == "pending"
        );
    }

    match /invites/{code} {
      allow read: if true;
      allow create, delete: if isAdmin();

      allow update: if isAdmin()
        || (
          signedIn()
          && resource.data.email == request.auth.token.email
          && resource.data.status == "pending"
          && request.resource.data.status == "accepted"
          && request.resource.data.uid == request.auth.uid
        );
    }

    match /notification_tokens/{uid} {
      allow create, read, update, delete: if signedIn() && request.auth.uid == uid;
      allow read, update, delete: if isAdmin();
    }

    match /taches/{id} {
      allow read: if isAdmin()
        || (isEmploye() && resource.data.employeeId == request.auth.uid);

      allow create, update, delete: if isAdmin();

      allow update: if isEmploye()
        && resource.data.employeeId == request.auth.uid
        && request.resource.data.employeeId == resource.data.employeeId;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
